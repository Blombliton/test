import pandas as pd
dataframe = pd.read_csv('test_ses.csv')
dataframe= dataframe.sort_values(by= "session_id" )[:10]

import matplotlib.pyplot as plt
fig, ax = plt.subplots()
dataframe.plot(kind = 'barh', x= "time2", y= "site2", ax = ax)
ax.set(title='График',
xlabel = 'Время', ylabel='Сайт')
ax.legend().set_visible(False)

avg = dataframe['site2'].mean()
competitors = dataframe[dataframe.site2 > avg]
ax.axhline(y=len(competitors)-0.5, color='green', label='Average', linestyle='-.', linewidth=4) 

fig.savefig('grafik.png', transparent=False, dpi=80, bbox_inches="tight")
